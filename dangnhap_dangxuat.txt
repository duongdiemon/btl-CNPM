import streamlit as st

import json
import os

# Thiáº¿t láº­p trang
st.set_page_config(
    page_title="Há»‡ Thá»‘ng ÄÄƒng Nháº­p",
    page_icon="âœ…",
    layout="wide"
)

# HÃ m Ä‘á»ƒ mÃ£ hÃ³a máº­t kháº©u
def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()

# HÃ m Ä‘á»ƒ lÆ°u dá»¯ liá»‡u ngÆ°á»i dÃ¹ng
def save_users():
    with open('users.json', 'w') as f:
        json.dump(st.session_state.users, f)

# HÃ m Ä‘á»ƒ táº£i dá»¯ liá»‡u ngÆ°á»i dÃ¹ng
def load_users():
    if os.path.exists('users.json'):
        with open('users.json', 'r') as f:
            return json.load(f)
    return {}

# Khá»Ÿi táº¡o session state
if 'users' not in st.session_state:
    st.session_state.users = load_users()
if 'logged_in' not in st.session_state:
    st.session_state.logged_in = False
if 'current_user' not in st.session_state:
    st.session_state.current_user = None

def login_page():
    with st.container():
        st.markdown('<div class="login-container">', unsafe_allow_html=True)
        st.markdown("### ðŸ” ÄÄƒng Nháº­p")
        with st.form("login_form"):
            username = st.text_input("ðŸ‘¤ TÃªn Ä‘Äƒng nháº­p:")
            password = st.text_input("ðŸ”‘ Máº­t kháº©u:", type="password")
            submit = st.form_submit_button("ÄÄƒng nháº­p")
            
            if submit:
                if username in st.session_state.users and st.session_state.users[username]['password'] == hash_password(password):
                    st.session_state.logged_in = True
                    st.session_state.current_user = username
                    st.success("âœ… ÄÄƒng nháº­p thÃ nh cÃ´ng!")
                    st.rerun()
                else:
                    st.error("âŒ TÃªn Ä‘Äƒng nháº­p hoáº·c máº­t kháº©u khÃ´ng Ä‘Ãºng!")
        st.markdown('</div>', unsafe_allow_html=True)

def register_page():
    with st.container():
        st.markdown('<div class="login-container">', unsafe_allow_html=True)
        st.markdown("### ðŸ“ ÄÄƒng KÃ½ TÃ i Khoáº£n")
        with st.form("register_form"):
            username = st.text_input("ðŸ‘¤ TÃªn Ä‘Äƒng nháº­p:")
            password = st.text_input("ðŸ”‘ Máº­t kháº©u:", type="password")
            confirm_password = st.text_input("ðŸ”„ XÃ¡c nháº­n máº­t kháº©u:", type="password")
            submit = st.form_submit_button("ÄÄƒng kÃ½")
            
            if submit:
                if username in st.session_state.users:
                    st.error("âŒ TÃªn Ä‘Äƒng nháº­p Ä‘Ã£ tá»“n táº¡i!")
                elif password != confirm_password:
                    st.error("âŒ Máº­t kháº©u xÃ¡c nháº­n khÃ´ng khá»›p!")
                elif len(password) < 6:
                    st.error("âŒ Máº­t kháº©u pháº£i cÃ³ Ã­t nháº¥t 6 kÃ½ tá»±!")
                else:
                    st.session_state.users[username] = {
                        'password': hash_password(password)
                    }
                    save_users()
                    st.success("âœ… ÄÄƒng kÃ½ thÃ nh cÃ´ng! Vui lÃ²ng Ä‘Äƒng nháº­p.")
        st.markdown('</div>', unsafe_allow_html=True)

def main():
    st.markdown("<h1 style='text-align: center; color: #1f77b4;'>âœ… Há»‡ Thá»‘ng ÄÄƒng Nháº­p</h1>", unsafe_allow_html=True)
    
    if not st.session_state.logged_in:
        tab1, tab2 = st.tabs(["ðŸ” ÄÄƒng nháº­p", "ðŸ“ ÄÄƒng kÃ½"])
        with tab1:

    # Header vá»›i thÃ´ng tin ngÆ°á»i dÃ¹ng

    with col3:
        if st.button("ðŸšª ÄÄƒng xuáº¥t"):
            st.session_state.logged_in = False
            st.session_state.current_user = None
            st.rerun()
